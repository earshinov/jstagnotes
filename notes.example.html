If you are viewing this page in Internet Explorer < 8, the date within this note's header will be positioned incorrectly. To correct this, you can:
  • In the source code of the note, place "date" block before the "title" block, but then the page will look incorrectly in other browsers (IE ≥ 8, Firefox etc.). This way is demostrated in the Test note 2
  • In the source code of the note, use "table" markup instead of "div" markup. The page will look nice in all supported browsers and… note title will be horizontally aligned even if it spans several lines. But the markup itself is not that pleasant. This way is demonstrated in the corresponding note.

02.06.2009

Можно использовать моноширинный текст прямо посреди строки, используя тэг <code>. Другой вариант - блок форматированного текста. Пример:

this.updateForSelectedTag = function(tag){
  $(".note:visible").each(function(){
    var $this = $(this);
    if (TagsCache.getNoteTags($this).indexOf(tag) === -1)
      $this.removeClass("selected").hide();
  });

  $("#notes .note_tag").filter(Predicates.hasText(tag)).addClass("chosen_tag");
};

this.updateForDeselectedTag = function(tag){
  if (Filter.isEmpty()){
    $("#notes .note_tag").removeClass("chosen_tag");
    $(".note").removeClass("selected").show();
    return;
  }

  $(".note:hidden").each(function(){
    if (noteSatisfiesFilter($(this)))
      $(this).show();
  });
  $(".selected").each(function(){
    if (noteSatisfiesFilter($(this)))
      $(this).removeClass("selected");
  });

  $("#notes .note_tag").filter(Predicates.hasText(tag)).removeClass("chosen_tag");
};

this.update = function(){
  $(".note").each(function(){
    $(this).removeClass("selected").toggle(noteSatisfiesFilter($(this)));
  });

  $("#notes .note_tag").each(function(){
    $(this).toggleClass("chosen_tag", Filter.isTagChosen($(this).text()));
  });
};
